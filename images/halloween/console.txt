document.getElementsByTagName('article')[2].className = 'halloween-map';

function insertTagAfter(parentTag, startingTagName, insertTag){
	parentName = parentTag.nodeName;
	parentHTML = parentTag.innerHTML;
	insertionIndex = parentHTML.indexOf('</'+startingTagName+'>') + 3 + startingTagName.length;
	firstPart = parentHTML.substring(0,insertionIndex);
	secondPart = parentHTML.substr(insertionIndex);

	insertName = insertTag.nodeName;
	insertHTML = '<' + insertName;

	attr = insertTag.attributes;
	for(i=0; i<attr.length;i++){
		insertHTML += ' ' + attr[i]['name'] + '=\"' + attr[i]['value'] + '\"';
	}

	insertHTML += '>' + insertTag.innerHTML + '</' + insertName + '>';

	parentTag.innerHTML = firstPart + insertHTML + secondPart;
	return parentTag;
}

art = document.getElementsByTagName('article')[2];
img1 = document.createElement('img');
img1.src="images/halloween/moonbats.png";
img1.className = "moonbats";
img2 = document.createElement('img');
img2.src="images/halloween/jacko.png";
img2.className = "jacko";
insertTagAfter(art, 'h2', img1);
insertTagAfter(art, 'p', img2);
